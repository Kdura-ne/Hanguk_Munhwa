CREATE DATABASE hangukmunhwa;

USE hangukmunhwa;


CREATE TABLE user(
userId INT NOT NULL AUTO_INCREMENT,
userNome VARCHAR(255) NOT NULL,
userEmail VARCHAR(255) NOT NULL UNIQUE,
userSenha VARCHAR(255) NOT NULL,
PRIMARY KEY (userId));

CREATE TABLE quiz(
quizId INT NOT NULL AUTO_INCREMENT,
particId INT NOT NULL,
quizData DATETIME DEFAULT CURRENT_TIMESTAMP,
quizScore INT NOT NULL,
PRIMARY KEY (quizId),
FOREIGN KEY (particId) REFERENCES user(userId) ON DELETE CASCADE ON UPDATE CASCADE);

ALTER DATABASE hangukmunhwa CHARACTER SET = utf8mb4 COLLATE = utf8mb4_unicode_ci;
ALTER TABLE user CONVERT TO CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
